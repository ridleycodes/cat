SET(LIBRARY_NAME crt)

### Generate CRT lib
IF(USE_OLD_CRT)
    SET(CRT_DEF "${CMAKE_CURRENT_SOURCE_DIR}/crtdll.def")
    SET(CRT_BASENAME "crtdll")
    IF(MSVC)
        SET(GLOBAL_COMPILE_FLAGS "${GLOBAL_COMPILE_FLAGS} /DUSE_OLD_CRT" PARENT_SCOPE)
    ELSE(MSVC)
        SET(GLOBAL_COMPILE_FLAGS "${GLOBAL_COMPILE_FLAGS} -DUSE_OLD_CRT" PARENT_SCOPE)
    ENDIF(MSVC)
ELSE(USE_OLD_CRT)
    SET(CRT_DEF "${CMAKE_CURRENT_SOURCE_DIR}/msvcrt.def")
    SET(CRT_BASENAME "msvcrt")
ENDIF(USE_OLD_CRT)

IF(MSVC)
    SET(LIB_EXT ".lib")
    SET(CRT_LIB "${CMAKE_CURRENT_BINARY_DIR}/${CRT_BASENAME}${LIB_EXT}")
    SET(LIB_CMD ${CMAKE_LIB_COMPILER} /nologo /def:"${CRT_DEF}" /out:"${CRT_LIB}" /machine:${MACHINE_ARCH})
ELSE(MSVC)
    SET(LIB_EXT ".a")
    SET(CRT_LIB "${CMAKE_CURRENT_BINARY_DIR}/lib${CRT_BASENAME}${LIB_EXT}")
    SET(LIB_CMD ${CMAKE_LIB_COMPILER} -d "${CRT_DEF}" -l "${CRT_LIB}" -m ${GNU_MACHINE_ARCH})
ENDIF(MSVC)

ADD_CUSTOM_COMMAND(
    OUTPUT "${CRT_LIB}"
    COMMAND ${LIB_CMD}
    DEPENDS "${CRT_DEF}"
)

ADD_CUSTOM_TARGET(${LIBRARY_NAME} DEPENDS "${CRT_LIB}")

LIST(APPEND EXECUTABLE_DEPENDENCIES ${LIBRARY_NAME})
SET(EXECUTABLE_DEPENDENCIES "${EXECUTABLE_DEPENDENCIES}" PARENT_SCOPE)
LIST(APPEND DLL_DEPENDENCIES ${LIBRARY_NAME})
SET(DLL_DEPENDENCIES "${DLL_DEPENDENCIES}" PARENT_SCOPE)

### Add CRT base
LIST(APPEND GLOBAL_ADDITIONAL_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/crt.c")
SET(GLOBAL_ADDITIONAL_SOURCES "${GLOBAL_ADDITIONAL_SOURCES}" PARENT_SCOPE)

LIST(APPEND GLOBAL_ADDITIONAL_LIBRARIES "${CRT_LIB}")
SET(GLOBAL_ADDITIONAL_LIBRARIES "${GLOBAL_ADDITIONAL_LIBRARIES}" PARENT_SCOPE)

### Define entry point
IF(MSVC)
    SET(EXECUTABLE_LINK_FLAGS "${EXECUTABLE_LINK_FLAGS} /entry:mainCRTStartup" PARENT_SCOPE)
    SET(DLL_COMPILE_FLAGS "${DLL_COMPILE_FLAGS} /DCRT_IS_DLL" PARENT_SCOPE)
    SET(DLL_LINK_FLAGS "${DLL_LINK_FLAGS} /entry:DllMainCRTStartup" PARENT_SCOPE)
ELSE(MSVC)
IF(MACHINE_ARCH STREQUAL "X86")
        SET(EXECUTABLE_LINK_FLAGS "${EXECUTABLE_LINK_FLAGS} -Wl,-e,_mainCRTStartup" PARENT_SCOPE)
        SET(DLL_LINK_FLAGS "${DLL_LINK_FLAGS} -Wl,-e,_DllMainCRTStartup@12" PARENT_SCOPE)
    ELSE(MACHINE_ARCH STREQUAL "X86")
        SET(EXECUTABLE_LINK_FLAGS "${EXECUTABLE_LINK_FLAGS} -Wl,-e,mainCRTStartup" PARENT_SCOPE)
        SET(DLL_LINK_FLAGS "${DLL_LINK_FLAGS} -Wl,-e,DllMainCRTStartup" PARENT_SCOPE)
    ENDIF(MACHINE_ARCH STREQUAL "X86")    
    SET(DLL_COMPILE_FLAGS "${DLL_COMPILE_FLAGS} -DCRT_IS_DLL" PARENT_SCOPE)
ENDIF(MSVC)
